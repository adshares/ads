304  for(uint32_t i=0; i<m_firstKeysLen; i+=2+32) {
305  char* firstkeys = m_firstVipKeys.getVipKeys();
306  uint16_t svid=*((uint16_t*)(&firstkeys[i+4]));
307  vipkeys[svid]=*((hash_s*)(&firstkeys[i+2+4]));
308  }
309 
310  int vok=0;
311 
312  for(const auto& s : m_signatures) {
313  auto it=vipkeys.find(s.svid);
314  if(it==vipkeys.end()) {
315  char hash[65];
316  hash[64]='\0';
317  ed25519_key2text(hash, s.signature, 32);
318  ELOG("ERROR vipkey (%d) not found %s\n", s.svid, hash);
319  continue;
320  }
321 
322  if(!ed25519_sign_open((const unsigned char*)(&m_receivedHeaders[m_numOfBlocks-1]), sizeof(header_t)-4, it->second.hash, s.signature)) {
323  vok++;
324  } else {
325  char hash[65];
326  hash[64]='\0';
327  ed25519_key2text(hash, s.signature, 32);
328  ELOG("ERROR vipkey (%d) failed %s\n", s.svid, hash);
329  }
330  }
331 
332  if(2*vok < (int)vipkeys.size()) {
333  ELOG("ERROR failed to confirm enough signature for header %08X (2*%d<%d)\n",
334  m_receivedHeaders[m_numOfBlocks-1].now, vok, (int)vipkeys.size());
335  return true;
336  } else {
337  ELOG("CONFIRMED enough signature for header %08X (2*%d>=%d)\n",
338  m_receivedHeaders[m_numOfBlocks-1].now, vok, (int)vipkeys.size());
339  }
