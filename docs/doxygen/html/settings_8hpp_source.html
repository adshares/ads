61  uint16_t crc16(const uint8_t* data_p, uint8_t length) {
62  uint8_t x;
63  uint16_t crc = 0x1D0F; //differet initial checksum !!!
64 
65  while(length--) {
66  x = crc >> 8 ^ *data_p++;
67  x ^= x>>4;
68  crc = (crc << 8) ^ ((uint16_t)(x << 12)) ^ ((uint16_t)(x <<5)) ^ ((uint16_t)x);
69  }
70  return crc;
71  }
