200 ErrorCodes::Code GetBlocksHandler::prepareResponse() {
201  uint32_t from = m_command->getBlockNumberFrom() - m_command->getBlockNumberFrom()%BLOCKSEC; //TODO replace 0 by the available beginning or last checkpoint
202  uint32_t to = m_command->getBlockNumberTo() - m_command->getBlockNumberTo()%BLOCKSEC;
203 
204  const uint32_t start = readStart();
205  if(!start) {
206  DLOG("ERROR, failed to read block start\n");
207  return ErrorCodes::Code::eFailedToReadBlockStart;
208  }
209  int vipTot = 0;
210  Block block;
211  header_t header{};
212 
213  if(!from || from==to) {
214  if (!from) {
215  from = start;
216  }
217  block.getData().ttime = from;
218 
219  if(!block.readDataFromHeaderFile()) {
220  DLOG("ERROR, failed to read header %08X\n", start);
221  return ErrorCodes::Code::eFailedToReadBlockAtStart;
222  }
223  header = block.getHeader();
224 
225  if(!prepareFirstVipKeys(block.getData().vipHash)) {
226  DLOG("ERROR, vip keys for start viphash %.64s will not be sent", block.getData().vipHash);
227  return ErrorCodes::Code::eCouldNotReadCorrectVipKeys;
228  }
229 
230  uint32_t length = m_firstVipKeys.getLength();
231  vipTot=length/(2+32);
232  DLOG("INFO, will send vip keys for start viphash %.64s [len:%d]\n", header.viphash, length);
233  }
234  else if(from < start) {
235  DLOG("ERROR, failed to read block %08X before start %08X\n", from, start);
236  return ErrorCodes::Code::eNoBlockInSpecifiedRange;
237  }
238  else {
239  block.getData().ttime = from-BLOCKSEC;
240  if(block.readDataFromHeaderFile()) {
241  vipTot=vipSize(block.getData().vipHash);
242  header = block.getHeader();
243  }
244  }
245 
246  if(!to) {

248  to-=to%BLOCKSEC-BLOCKSEC;
249  }
250 
251  if(from>to) {
252  DLOG("ERROR, no block between %08X and %08X\n", from, to);
253  }

256  if(m_serversHeaders.size() > 0) {

258  loadLastBlockSignatures(block.getData().ttime);
259  if(m_newviphash) {

261  }
262  }
263  else {
264  DLOG("ERROR, failed to provide blocks from %08X to %08X\n", from, block.getData().ttime);
265  }
266  }
