152 bool GetTransaction::checkMessageHash(char *data) {
153  uint16_t svid= m_data.info.dst_node;
154  uint32_t msid= m_data.info.node_msgid;
155  uint16_t tnum= m_data.info.position;
156  if (!m_responseHeader.path) {
157  m_responseError = ErrorCodes::Code::eGotEmptyBlock;
158  ELOG("ERROR, got empty block for txid %04X:%08X:%04X\n",svid,msid,tnum);
159  return false;
160  }
161  if (m_responseHeader.node != svid
162  || m_responseHeader.msid != msid
163  || m_responseHeader.tnum != tnum) {
164  m_responseError = ErrorCodes::Code::eIncorrectTransaction;
165  ELOG("ERROR, got wrong transaction %04X%08X%04X for txid %04X:%08X:%04X\n",
166  m_responseHeader.node, m_responseHeader.msid, m_responseHeader.tnum, svid, msid, tnum);
167  return false;
168  }
169  Helper::Servers block;
170  block.setNow(m_responseHeader.path);
171  if (!block.loadNowhash()) {
172  m_responseError = ErrorCodes::Code::eFailedToLoadHash;
173  return false;
174  }
175 
176  hash_t nhash;
177  uint16_t* hashsvid=(uint16_t*)nhash;
178  uint32_t* hashmsid=(uint32_t*)(&nhash[4]);
179  uint16_t* hashtnum=(uint16_t*)(&nhash[8]);
180  SHA256_CTX sha256;
181  SHA256_Init(&sha256);
182  SHA256_Update(&sha256, data, m_responseHeader.len);
183  SHA256_Final(nhash,&sha256);
184  *hashsvid^=svid;
185  *hashmsid^=msid;
186  *hashtnum^=tnum;
187 
188  if(memcmp(nhash, &m_responseHashes[0], 32)) {
189  ELOG("ERROR, failed to confirm first hash for txid %04X:%08X:%04X\n",svid,msid,tnum);
190  m_responseError = ErrorCodes::Code::eHashMismatch;
191  return false;
192  }
193 
194  hashtree tree;
195  tree.hashpathrun(nhash, m_responseHashes);
196  if(memcmp(nhash, block.getNowHash(), 32)) {
197  ELOG("ERROR, failed to confirm nowhash for txid %04X:%08X:%04X\n",svid,msid,tnum);
198  m_responseError = ErrorCodes::Code::eHashMismatch;
199  return false;
200  }
201  return true;
202 }
