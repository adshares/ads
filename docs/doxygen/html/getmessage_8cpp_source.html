98  ELOG("GetMessage reading error\n");

100  }
101 
102  if (m_responseError) {
103  return true;
104  }
105 
106  // msgTypeLength contains type (1B) and length (3B)
107  uint32_t msgTypeLength = 0;
108  if (!netClient.readData((int32_t*)&msgTypeLength, sizeof(msgTypeLength))) {
109  ELOG("GetMessage reading length error\n");
110  return false;
111  }
112  uint32_t msgLength = msgTypeLength >> 8;
113  if (!msgTypeLength || msgLength < message::data_offset) {
114  m_responseError = ErrorCodes::Code::eBadLength;
115  return true;
116  }
117  uint8_t msgType = msgTypeLength & 0xFF;
118  if (msgType != MSGTYPE_MSG) {
119  m_responseError = ErrorCodes::Code::eIncorrectTransaction;
120  return true;
121  }
122 
123  char buffer[msgLength];
124  memcpy(buffer, &msgTypeLength, 4);
125  if (!netClient.readData(buffer + 4, msgLength - 4)) {
126  ELOG("GetMessage reading error\n");
127  return false;
128  }
129 
130  m_responseMsg.reset(new message(msgLength, (uint8_t*)buffer));
131
