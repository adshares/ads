153 ErrorCodes::Code GetAccounts::prepareResponse(uint32_t lastPath, uint32_t lastUsers) {
154  ErrorCodes::Code errorCode = ErrorCodes::Code::eNone;
155  uint32_t destBank = this->getDestBankId();
156  char filename[64];
157  int fd, ud;
158 
159  sprintf(filename,"usr/%04X.dat", destBank);
160  fd = open(filename,O_RDONLY);
161  if (fd < 0) {
162  errorCode = ErrorCodes::Code::eBankNotFound;
163  } else {
164  sprintf(filename,"blk/%03X/%05X/und/%04X.dat",lastPath>>20,lastPath&0xFFFFF, destBank);
165  ud = open(filename,O_RDONLY);
166  if(ud < 0) {
167  errorCode = ErrorCodes::Code::eUndoNotFound;
168  close(fd);
169  }
170  }
171 
172  if (!errorCode) {
173  m_responseBufferLength = lastUsers * sizeof(user_t);
174  m_responseBuffer = new unsigned char[m_responseBufferLength];
175  uint8_t *dp = m_responseBuffer;
176  for(uint32_t user=0; user<lastUsers; user++,dp+=sizeof(user_t)) {
177  user_t u;
178  u.msid=0;
179  read(fd,dp,sizeof(user_t));
180  read(ud,&u,sizeof(user_t));
181  if(u.msid!=0) {
182  memcpy(dp,&u,sizeof(user_t));
183  }
184  }
185  close(ud);
186  close(fd);
187  }
188  return errorCode;
189 }
