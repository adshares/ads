360  if(!tree.hashes.size()) {
361  hashes_size=0;

364  data=(uint8_t*)std::realloc(data,total);

372  }

880  }
881  }

894  ELOG("DBL bad message hash_tree_fast 1 failed\n");
895  return(-1);

899  ed25519_sign_open2(dblmsha,32,sigh2,32,svpk,data2+4));
900  }
901  ELOG("DBL message illegal type\n");
902  return(-1);
903  }
