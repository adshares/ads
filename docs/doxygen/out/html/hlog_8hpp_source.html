38  hlog(boost::property_tree::ptree& pt,char* filename) : //log,filename
39  total(0),
40  data(NULL),
41  fd(-1)
42  { SHA256_Init(&sha256);

44  if(fd<0){
45  pt.put("error_read","true");
46  return;}
47  uint8_t type;
48  boost::property_tree::ptree logs;
49  while(read(fd,&type,1)>0){
50  boost::property_tree::ptree log;
51  switch(type){
52  case(HLOG_USO): read_uso(log); break;
53  case(HLOG_UOK): read_uok(log); break;
54  case(HLOG_USR): read_usr(log); break;
55  case(HLOG_BKY): read_bky(log); break;
56  case(HLOG_SBS): read_sbs(log); break;
57  case(HLOG_UBS): read_ubs(log); break;
58  case(HLOG_BNK): read_bnk(log); break;
59  case(HLOG_GET): read_get(log); break;
60  default:
61  pt.add_child("logs",logs);
62  pt.put("error_parse","true");
63  return;}
64  logs.push_back(std::make_pair("",log));}
65  char hash[65]; hash[64]='\0';
66  hash_t fin;
67  finish(fin,total);
68  ed25519_key2text(hash,fin,32);
